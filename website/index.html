hsp-example.ts 
import { HSPEngine, CriticalityLevel } from '@psan/protocol';

const hsp = new HSPEngine({
  defaultLevel: CriticalityLevel.MEDIUM,
  escalationRules: [
    { 
      condition: (action) => action.value > 1000, 
      level: CriticalityLevel.HIGH 
    }
  ]
});

// Process an agent action
const result = await hsp.process({
  agentId: 'financial-agent-001',
  action: 'transfer',
  value: 5000,
  target: 'vendor-account'
});

if (result.requiresApproval) {
  // Human stays in control
  await requestHumanApproval(result);
}
